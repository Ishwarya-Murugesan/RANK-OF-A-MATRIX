#Program to find the rank of a matrix.
import numpy as np

# Define the matrix
A = np.array([[5, -3, -10],
              [2,  2, -3],
              [-3, -1, 5]])

# Compute rank
rank = np.linalg.matrix_rank(A)

print( rank)
